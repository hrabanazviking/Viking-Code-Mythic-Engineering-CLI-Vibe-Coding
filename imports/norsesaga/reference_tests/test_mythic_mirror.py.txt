import logging
import pytest
from systems.mythic_mirror import MythicMirror, ARCHETYPE_MAP, SIGNAL_KEYWORDS

# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------


def _force_archetype(mm: MythicMirror, signal_key: str, count: int = 10) -> None:
    """
    Directly set the named signal high enough to guarantee dominance > 0.25,
    then trigger update_from_turn with empty text so the archetype check runs.
    """
    mm.signals = {k: 0 for k in mm.signals}
    mm.signals[signal_key] = count
    mm.update_from_turn("", "")


# ---------------------------------------------------------------------------
# 1. Initialisation defaults
# ---------------------------------------------------------------------------


def test_init_defaults():
    """All signals start at 0 and archetype is 'wanderer'."""
    mm = MythicMirror()
    assert mm.archetype == "wanderer"
    assert mm.archetype_description == "One who seeks what lies beyond the horizon"
    assert mm.titles == []
    assert mm._previous_archetype == "wanderer"
    assert set(mm.signals.keys()) == {"mercy", "conflict", "exploration", "loyalty", "mystery", "change"}
    assert all(v == 0 for v in mm.signals.values())


# ---------------------------------------------------------------------------
# 2. Conflict keywords increment conflict signal
# ---------------------------------------------------------------------------


def test_conflict_keywords_increment_conflict_signal():
    """Player input containing 'attack' increments signals['conflict'] by 1."""
    mm = MythicMirror()
    before = mm.signals["conflict"]
    mm.update_from_turn("I attack the wolf", "")
    assert mm.signals["conflict"] == before + 1


def test_multiple_keyword_matches_counted_individually():
    """Each matching keyword increments its signal separately."""
    mm = MythicMirror()
    mm.update_from_turn("I fight and attack and charge", "")
    assert mm.signals["conflict"] == 3  # 'fight', 'attack', 'charge'


def test_narrative_text_also_scanned():
    """Keywords in the narrative argument are counted too."""
    mm = MythicMirror()
    mm.update_from_turn("", "The warrior strikes with rage and battle fury")
    # 'strike', 'rage', 'battle' → conflict += 3
    assert mm.signals["conflict"] == 3


def test_update_ignores_case():
    """Keyword matching is case-insensitive (input is lowercased before scan)."""
    mm = MythicMirror()
    mm.update_from_turn("I ATTACK", "")
    assert mm.signals["conflict"] >= 1


# ---------------------------------------------------------------------------
# 3. Mercy keywords increment mercy signal
# ---------------------------------------------------------------------------


def test_mercy_keywords_increment_mercy_signal():
    """Input containing 'spare' and 'heal' increments mercy by 2."""
    mm = MythicMirror()
    mm.update_from_turn("I spare the warrior and heal his wounds", "")
    assert mm.signals["mercy"] == 2


# ---------------------------------------------------------------------------
# 4. Archetype shifts when dominant signal exceeds threshold
# ---------------------------------------------------------------------------


def test_archetype_shifts_when_dominant_signal_exceeds_threshold():
    """
    After pure conflict input, conflict dominance = 1.0 > 0.25,
    archetype shifts to 'stormbringer'.
    """
    mm = MythicMirror()
    mm.update_from_turn("I fight the enemy", "")
    assert mm.archetype == "stormbringer"


def test_single_dominant_signal_immediately_shifts():
    """One turn with a single signal makes that signal 100% dominant → shift."""
    mm = MythicMirror()
    mm.update_from_turn("I explore and wander and search", "")
    assert mm.archetype == "wanderer"  # exploration maps to 'wanderer' — already default
    # Switch to a non-default archetype
    mm2 = MythicMirror()
    mm2.update_from_turn("I rune a vision and dream of spirits", "")
    assert mm2.archetype == "seer"


# ---------------------------------------------------------------------------
# 5. Archetype does not shift below threshold
# ---------------------------------------------------------------------------


def test_archetype_does_not_shift_below_threshold():
    """
    With signals spread evenly, no single signal dominates > 25%,
    archetype stays 'wanderer'.
    """
    mm = MythicMirror()
    mm.signals = {"mercy": 1, "conflict": 1, "exploration": 1,
                  "loyalty": 1, "mystery": 0, "change": 0}
    # total=4, max=1/4=0.25, NOT strictly > 0.25 → no shift
    mm.update_from_turn("", "")
    assert mm.archetype == "wanderer"


def test_dominance_boundary_at_exactly_025_does_not_shift():
    """
    Dominance exactly equal to 0.25 does NOT trigger a shift
    (condition is strictly greater than).
    """
    mm = MythicMirror()
    # conflict=1 out of total=4 → dominance=0.25 (not > 0.25)
    mm.signals = {"mercy": 1, "conflict": 1, "exploration": 1, "loyalty": 1,
                  "mystery": 0, "change": 0}
    mm.update_from_turn("", "")
    assert mm.archetype == "wanderer"


# ---------------------------------------------------------------------------
# 6. Archetype description matches ARCHETYPE_MAP entry
# ---------------------------------------------------------------------------


def test_archetype_description_matches_archetype():
    """
    After an archetype shift, archetype_description matches ARCHETYPE_MAP
    for the new archetype key.
    """
    mm = MythicMirror()
    mm.update_from_turn("fight attack charge kill battle slay", "")
    assert mm.archetype == "stormbringer"
    assert mm.archetype_description == ARCHETYPE_MAP["conflict"][1]
    assert mm.archetype_description == "One who draws battle like thunder draws lightning"


def test_all_archetype_map_entries_reachable():
    """Every signal in ARCHETYPE_MAP can be triggered to produce its archetype."""
    for signal_key, (expected_archetype, expected_desc) in ARCHETYPE_MAP.items():
        mm = MythicMirror()
        _force_archetype(mm, signal_key)
        assert mm.archetype == expected_archetype, (
            f"Signal '{signal_key}' should produce archetype '{expected_archetype}' "
            f"but got '{mm.archetype}'"
        )
        assert mm.archetype_description == expected_desc


# ---------------------------------------------------------------------------
# 7. _previous_archetype logged on shift
# ---------------------------------------------------------------------------


def test_previous_archetype_updated_on_shift():
    """_previous_archetype holds the old archetype after a shift."""
    mm = MythicMirror()
    assert mm._previous_archetype == "wanderer"
    mm.update_from_turn("I fight the enemy", "")
    assert mm.archetype == "stormbringer"
    assert mm._previous_archetype == "wanderer"


def test_previous_archetype_logged_on_shift(caplog):
    """logger.info is called with the shift message when archetype changes."""
    mm = MythicMirror()
    with caplog.at_level(logging.INFO, logger="systems.mythic_mirror"):
        mm.update_from_turn("I fight the enemy", "")
    assert any("shifted" in record.message for record in caplog.records), (
        "Expected a 'shifted' log message from the Mythic Mirror"
    )
    assert any("wanderer" in record.message and "stormbringer" in record.message
               for record in caplog.records)


def test_no_log_when_archetype_unchanged(caplog):
    """No shift log when the archetype resolves to the same value it already is."""
    mm = MythicMirror()
    # 'exploration' already maps to 'wanderer' — the same as the default
    with caplog.at_level(logging.INFO, logger="systems.mythic_mirror"):
        mm.update_from_turn("I explore and wander", "")
    shift_logs = [r for r in caplog.records if "shifted" in r.message]
    assert shift_logs == [], "Should not log a shift when archetype is unchanged"

def test_title_list_capped_at_five():
    """Adding 6 unique titles keeps only the last 5."""
    mm = MythicMirror()
    for i in range(6):
        mm.add_title(f"Title_{i}")
    assert len(mm.titles) == 5
    assert "Title_0" not in mm.titles
    assert "Title_5" in mm.titles


def test_title_list_preserves_order():
    """The kept titles are the most recently added (last 5)."""
    mm = MythicMirror()
    for i in range(7):
        mm.add_title(f"Title_{i}")
    assert mm.titles == ["Title_2", "Title_3", "Title_4", "Title_5", "Title_6"]

    """build_context() shows at most the last 3 titles."""
    mm = MythicMirror()
    for t in ["Fjord-Born", "Ice-Hand", "Raven-Eye", "Wolf-Pelt", "Storm-Axe"]:
        mm.add_title(t)
    ctx = mm.build_context()
    # The 5 unique titles fit within max_active=5; last-3 shown are Raven-Eye, Wolf-Pelt, Storm-Axe
    assert "Raven-Eye" in ctx
    assert "Wolf-Pelt" in ctx
    assert "Storm-Axe" in ctx
    # Older titles beyond the last-3 window must be absent
    assert "Fjord-Born" not in ctx
    assert "Ice-Hand" not in ctx

# ---------------------------------------------------------------------------
# 14. Serialisation round-trip
# ---------------------------------------------------------------------------


def test_serialisation_round_trip():
    """to_dict() → from_dict() preserves archetype, signals, and titles exactly."""
    mm = MythicMirror()
    mm.update_from_turn("fight attack", "")
    mm.add_title("Oathbreaker")
    snapshot = mm.to_dict()

    assert set(snapshot.keys()) == {"archetype", "archetype_description", "titles", "signals"}

    mm2 = MythicMirror()
    mm2.from_dict(snapshot)
    assert mm2.archetype == mm.archetype
    assert mm2.archetype_description == mm.archetype_description
    assert mm2.titles == mm.titles
    assert mm2.signals == mm.signals


def test_to_dict_keys():
    """to_dict() exposes exactly the expected four keys."""
    expected = {"archetype", "archetype_description", "titles", "signals"}
    assert set(MythicMirror().to_dict().keys()) == expected


# ---------------------------------------------------------------------------
# 15. from_dict with missing 'signals' key falls back to all-zero dict
# ---------------------------------------------------------------------------


def test_from_dict_missing_signals_defaults_to_zeros():
    """from_dict() with no 'signals' key falls back to the all-zero dict."""
    mm = MythicMirror()
    mm.from_dict({"archetype": "stormbringer",
                  "archetype_description": "test desc",
                  "titles": ["Iron-Hand"]})
    assert all(v == 0 for v in mm.signals.values())
    assert set(mm.signals.keys()) == {"mercy", "conflict", "exploration",
                                       "loyalty", "mystery", "change"}


def test_from_dict_restores_non_default_archetype():
    """from_dict() correctly restores a non-default archetype."""
    mm = MythicMirror()
    mm.from_dict({"archetype": "seer",
                  "archetype_description": "One who walks between worlds",
                  "titles": [],
                  "signals": {"mercy": 0, "conflict": 0, "exploration": 0,
                               "loyalty": 0, "mystery": 5, "change": 0}})
    assert mm.archetype == "seer"
    assert mm.signals["mystery"] == 5


def test_from_dict_empty_leaves_defaults():
    """from_dict({}) does not raise; archetype remains 'wanderer'."""
    mm = MythicMirror()
    mm.from_dict({})
    assert mm.archetype == "wanderer"
    assert mm.titles == []


def test_from_dict_none_leaves_defaults():
    """from_dict(None) does not raise; state stays at defaults."""
    mm = MythicMirror()
    mm.from_dict(None)
    assert mm.archetype == "wanderer"
    assert mm.titles == []

def test_archetype_map_has_six_entries():
    """ARCHETYPE_MAP covers all six behavioral signals."""

        "exploration", "loyalty", "conflict", "mystery", "mercy", "change"
    }


def test_signal_keywords_has_six_groups():
    """SIGNAL_KEYWORDS covers all six signal types."""
    assert set(SIGNAL_KEYWORDS.keys()) == set(ARCHETYPE_MAP.keys())


def test_each_keyword_group_non_empty():
    """Every keyword group has at least one keyword."""
    for signal_type, keywords in SIGNAL_KEYWORDS.items():
        assert len(keywords) > 0, f"Keyword group '{signal_type}' is empty"
        assert all(isinstance(kw, str) for kw in keywords)
